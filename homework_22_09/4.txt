Проще привести листинг кода:

#include <string>

int f(int i); // предварительное объявление

int main()
{
    const double pi = 3.14; // ОК, мы объявили и определили переменную
    int i = f(2); // OK, f объявляется заранее
    C obj; // ошибка! C еще не объявлен.
    std::string str; // OK, std::string объявлен в заголовке <string>
    j = 0; // ошибка! Тип не указан. Не было объявления, не получится определить переменную
    auto k = 0; // OK, тип, выведенный компилятором как int.
}

int f(int i) // определение функции
{
    return i + 42;
